cmake_minimum_required(VERSION 3.20)
add_executable(01_cmd_queue main.cpp)
target_include_directories(01_cmd_queue PRIVATE ${GRAPHICS_COMMON_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../..)
target_link_libraries(01_cmd_queue PRIVATE user32 d3d12 dxgi)
set_target_properties(01_cmd_queue PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
if(NOT MSVC)
    target_link_options(01_cmd_queue PRIVATE -municode)
endif()
