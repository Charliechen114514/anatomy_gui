cmake_minimum_required(VERSION 3.20)
add_executable(02_resource_heap main.cpp)
target_include_directories(02_resource_heap PRIVATE ${GRAPHICS_COMMON_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../..)

if(NOT MSVC)
    target_link_options(02_resource_heap PRIVATE -municode)
endif()
