cmake_minimum_required(VERSION 3.20)
add_executable(01_owner_draw main.cpp)
target_include_directories(01_owner_draw PRIVATE ${GRAPHICS_COMMON_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../..)
target_link_libraries(01_owner_draw PRIVATE user32 gdi32 comctl32)
set_target_properties(01_owner_draw PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
if(NOT MSVC)
    target_link_options(01_owner_draw PRIVATE -municode)
endif()
