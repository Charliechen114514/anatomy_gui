cmake_minimum_required(VERSION 3.20)
project(02_transform_matrix LANGUAGES CXX)
add_executable(02_transform_matrix main.cpp)
target_include_directories(02_transform_matrix PRIVATE ${GRAPHICS_COMMON_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../..)
target_link_libraries(02_transform_matrix PRIVATE user32 gdi32 gdiplus)
set_target_properties(02_transform_matrix PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
if(NOT MSVC)
    target_link_options(02_transform_matrix PRIVATE -municode)
endif()
