cmake_minimum_required(VERSION 3.20)
project(01_gdiplus_hello LANGUAGES CXX)
add_executable(01_gdiplus_hello main.cpp)
target_include_directories(01_gdiplus_hello PRIVATE ${GRAPHICS_COMMON_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../..)
target_link_libraries(01_gdiplus_hello PRIVATE user32 gdi32 gdiplus)
set_target_properties(01_gdiplus_hello PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
if(NOT MSVC)
    target_link_options(01_gdiplus_hello PRIVATE -municode)
endif()
