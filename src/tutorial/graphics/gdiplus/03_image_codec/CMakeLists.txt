cmake_minimum_required(VERSION 3.20)
project(03_image_codec LANGUAGES CXX)
add_executable(03_image_codec main.cpp)
target_include_directories(03_image_codec PRIVATE ${GRAPHICS_COMMON_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../..)
target_link_libraries(03_image_codec PRIVATE user32 gdi32 gdiplus comctl32 shlwapi)
set_target_properties(03_image_codec PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
if(NOT MSVC)
    target_link_options(03_image_codec PRIVATE -municode)
endif()
