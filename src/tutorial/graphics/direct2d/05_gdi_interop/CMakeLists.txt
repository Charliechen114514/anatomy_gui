cmake_minimum_required(VERSION 3.20)
add_executable(05_gdi_interop main.cpp)
target_include_directories(05_gdi_interop PRIVATE ${GRAPHICS_COMMON_DIR} ${CMAKE_CURRENT_SOURCE_DIR}/../..)
target_link_libraries(05_gdi_interop PRIVATE user32 gdi32 d2d1 dwrite)
set_target_properties(05_gdi_interop PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
if(NOT MSVC)
    target_link_options(05_gdi_interop PRIVATE -municode)
endif()
